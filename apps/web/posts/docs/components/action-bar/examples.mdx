## Responsive page header

```jsx
function Demo() {
  const toggle = useToggle();
  const items = ["Documentation", "About us", "Pricing"];

  return (
    <>
      <ActionBar position="top" size="large">
        <View gap={4} align="center" direction="row">
          <View.Item grow={{ s: true, m: false }}>
            <View
              width="40px"
              height="40px"
              backgroundColor="neutral"
              borderRadius="small"
            />
          </View.Item>
          <Hidden hide={{ s: true, m: false }}>
            <View.Item grow key="menu">
              <View direction="row" gap={1}>
                {items.map((item) => (
                  <Button variant="ghost" key={item}>
                    {item}
                  </Button>
                ))}
              </View>
            </View.Item>
          </Hidden>
          <Button.Aligner position="end">
            <View gap={2} direction="row">
              <Button variant="ghost" icon={IconHeart} />
              <Hidden hide={{ s: false, m: true }}>
                <Button
                  variant="ghost"
                  icon={IconMenu}
                  onClick={toggle.activate}
                />
              </Hidden>
            </View>
          </Button.Aligner>
        </View>
      </ActionBar>
      <Modal
        position="bottom"
        active={toggle.active}
        onClose={toggle.deactivate}
        closeAriaLabel="Close menu"
        padding={1}
      >
        {items.map((item) => (
          <MenuItem key={item} roundedCorners>
            {item}
          </MenuItem>
        ))}
      </Modal>
    </>
  );
}
```
